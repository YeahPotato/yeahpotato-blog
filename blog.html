AJAX异步请求

          function ajax(options){
              options=options||{};
              options.data=options.data||{};
              options.type=options.type||'get';
              options.timeout=options.timeout||0;
              options.data.t=Math.random();
              
              var arr=[];
              for(var name in options.data){
                arr.push(name+'='+encodeURIComponent(options.data[name]));  
              } 
              var str=arr.join('&');
              
              
              if(window.XMLHttpRequest){
                var oAjax=new XMLHttpRequest();   
              }else{
                var oAjax=new ActiveXObject('Microsoft.XMLHTTP');   
              }
              
              if(options.type=='get'){
                oAjax.open('get',options.url+'?'+str,true);
                oAjax.send(); 
              }else{
                oAjax.open('post',options.url,true);
                oAjax.setRequestHeader('Content-Type','application/x-www-form-urlencoded'); 
                oAjax.send(str);
              }
              oAjax.onreadystatechange=function(){
                if(oAjax.readyState==4){
                  clearTimeout(timer);
                  if(oAjax.status>=200 && oAjax.status<300||oAjax.status==304){
                    options.success && options.sucess(oAjax.responseText);  
                  }else{
                    options.error && options.error(oAjax.status); 
                  }
                } 
              };
              if(options.timeout){
                var timer=setTimeout(function(){
                  oAjax.abort();  
                },options.timeout); 
              }
            }
